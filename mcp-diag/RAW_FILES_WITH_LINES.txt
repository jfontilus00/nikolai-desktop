   23: function splitArgs(input: string): string[] {
   24:   const s = String(input || "").trim();
   25:   if (!s) return [];
   26:   const out: string[] = [];
   27:   let cur = "";
   28:   let q: '"' | "'" | null = null;
   29: 
   30:   for (let i = 0; i < s.length; i++) {
   31:     const ch = s[i];
   32: 
   33:     if (q) {
   34:       if (ch === "\\" && i + 1 < s.length) {
   35:         // keep escaped next char
   36:         cur += s[i + 1];
   37:         i++;
   38:         continue;
   39:       }
   40:       if (ch === q) {
   41:         q = null;
   42:         continue;
   43:       }
   44:       cur += ch;
   45:       continue;
   46:     }
   47: 
   48:     if (ch === '"' || ch === "'") {
   49:       q = ch as any;
   50:       continue;
   51:     }
   52: 
   53:     if (/\s/.test(ch)) {
   54:       if (cur.length) {
   55:         out.push(cur);
   56:         cur = "";
   57:       }
   58:       continue;
   59:     }
   60: 
   61:     cur += ch;
   62:   }
   63: 
   64:   if (cur.length) out.push(cur);
   65:   return out;
   66: }

  170:       if let McpState::Connected { ref client, .. } = *state {
  171:         // Fail all pending requests
  172:         fail_all_pending(client, "Buffer overflow").await;
  173:       }
  174:       *state = McpState::Error {
  175:         message: "Buffer overflow: MCP server sent too much data".to_string(),
  176:         at: now_millis(),
  177:       };
  178:       emit_state_change(&app, "error", Some("Buffer overflow".to_string()));
  179:       return;
  180:     }
  181: 
  182:     loop {
  183:       let s = match std::str::from_utf8(&buf) {
  184:         Ok(s) => s,
  185:         Err(_) => break,
  186:       };
  187: 
  188:       // Support LSP-style Content-Length framing
  189:       if s.starts_with("Content-Length:") {
  190:         if let Some(hdr_end) = s.find("\r\n\r\n") {
  191:           let headers = &s[..hdr_end];
  192:           let mut len: usize = 0;
  193: 
  194:           for line in headers.lines() {
  195:             if let Some(v) = line.strip_prefix("Content-Length:") {
  196:               len = v.trim().parse::<usize>().unwrap_or(0);
  197:             }
  198:           }
  199: 
  200:           let total_needed = hdr_end + 4 + len;
  201:           if buf.len() < total_needed {
  202:             break;
  203:           }
  204: 
  205:           let json_bytes = buf[(hdr_end + 4)..(hdr_end + 4 + len)].to_vec();
  206:           let json_str = String::from_utf8_lossy(&json_bytes).to_string();
  207:           buf.drain(0..total_needed);
  208: 
  209:           handle_message(&json_str, &client, &app).await;
  210:           continue;
  211:         }
  212:         break;
  213:       }
  214: 
  215:       // Newline-delimited JSON
  216:       if let Some(pos) = s.find('\n') {
  217:         let line = s[..pos].trim().to_string();
  218:         buf.drain(0..pos + 1);
  219: 
  220:         if !line.is_empty() {
  221:           handle_message(&line, &client, &app).await;
  222:         }
  223:         continue;
  224:       }
  225: 
  226:       break;
  227:     }
  228:   }
  229: 
  230:   // EOF reached - normal exit
  231:   eprintln!("[mcp] Reader loop exiting (EOF)");
  232: }

  234: async fn notify_with_client(client: Arc<McpClient>, method: &str, params: Value) -> Result<(), String> {
  235:   let msg = json!({
  236:     "jsonrpc": "2.0",
  237:     "method": method,
  238:     "params": params
  239:   });
  240:   let text = msg.to_string() + "\n";
  241: 
  242:   let mut stdin = client.stdin.lock().await;
  243:   stdin
  244:     .write_all(text.as_bytes())
  245:     .await
  246:     .map_err(|e| format!("stdin write failed: {}", e))?;
  247:   stdin.flush().await.map_err(|e| format!("stdin flush failed: {}", e))?;
  248:   Ok(())
  249: }

  251: async fn request_with_client(client: Arc<McpClient>, method: &str, params: Value) -> Result<Value, String> {
  252:   let id = client.next_id.fetch_add(1, Ordering::SeqCst);
  253: 
  254:   let req = json!({
  255:     "jsonrpc":"2.0",
  256:     "id": id,
  257:     "method": method,
  258:     "params": params
  259:   });
  260: 
  261:   let (tx, rx) = oneshot::channel::<Result<Value, Value>>();
  262:   {
  263:     let mut pending = client.pending.lock().await;
  264:     pending.insert(id, tx);
  265:   }
  266: 
  267:   {
  268:     let mut stdin = client.stdin.lock().await;
  269:     let line = req.to_string() + "\n";
  270:     stdin.write_all(line.as_bytes()).await.map_err(|e| e.to_string())?;
  271:     stdin.flush().await.map_err(|e| e.to_string())?;
  272:   }
  273: 
  274:   match timeout(Duration::from_secs(MCP_TIMEOUT_SECS), rx).await {
  275:     Ok(Ok(Ok(v))) => Ok(v),
  276:     Ok(Ok(Err(e))) => Err(format!("MCP error: {}", e)),
  277:     Ok(Err(_)) => Err("MCP response channel closed".to_string()),
  278:     Err(_) => Err("MCP timeout".to_string()),
  279:   }
  280: }

  282: async fn request(method: &str, params: Value) -> Result<Value, String> {
  283:   let state = ensure_state();
  284:   let client = {
  285:     let guard = state.lock().await;
  286:     match &*guard {
  287:       McpState::Connected { client, .. } => client.clone(),
  288:       _ => return Err("MCP not connected".to_string()),
  289:     }
  290:   };
  291:   request_with_client(client, method, params).await
  292: }

  321: #[tauri::command]
  322: pub async fn mcp_connect(
  323:   cfg: McpConnectConfig,
  324:   app: AppHandle,
  325: ) -> Result<McpStatus, String> {
  326:   let state = ensure_state();
  327:   let cfg_summary = format!("{} {}", cfg.command, cfg.args.join(" "));
  328:   
  329:   // Atomic check-and-transition under ONE lock
  330:   let transition_result = {
  331:     let mut guard = state.lock().await;
  332:     match &*guard {
  333:       McpState::Disconnected | McpState::Error { .. } => {
  334:         // Can start new connection
  335:         *guard = McpState::Connecting {
  336:           started_at: now_millis(),
  337:           cfg_summary: cfg_summary.clone(),
  338:         };
  339:         emit_state_change(&app, "connecting", Some(cfg_summary.clone()));
  340:         Ok(())
  341:       }
  342:       McpState::Connecting { .. } | McpState::Connected { .. } => {
  343:         // Already connecting or connected
  344:         Err("Already connected or connecting".to_string())
  345:       }
  346:     }
  347:   };
  348: 
  349:   // If transition failed, return current status
  350:   if let Err(_) = transition_result {
  351:     return Ok(mcp_status().await);
  352:   }
  353: 
  354:   // Validate command
  355:   if cfg.command.trim().is_empty() {
  356:     let mut guard = state.lock().await;
  357:     *guard = McpState::Error {
  358:       message: "Command is empty".to_string(),
  359:       at: now_millis(),
  360:     };
  361:     emit_state_change(&app, "error", Some("Command is empty".to_string()));
  362:     return Err("Command is empty".to_string());
  363:   }
  364: 
  365:   // Spawn child process
  366:   let mut cmd = Command::new(&cfg.command);
  367:   cmd.args(&cfg.args)
  368:     .stdin(Stdio::piped())
  369:     .stdout(Stdio::piped())
  370:     .stderr(Stdio::inherit());
  371: 
  372:   if let Some(cwd) = &cfg.cwd {
  373:     if !cwd.trim().is_empty() {
  374:       cmd.current_dir(cwd);
  375:     }
  376:   }
  377: 
  378:   let mut child = match cmd.spawn() {
  379:     Ok(c) => c,
  380:     Err(e) => {
  381:       let mut guard = state.lock().await;
  382:       *guard = McpState::Error {
  383:         message: format!("spawn failed: {}", e),
  384:         at: now_millis(),
  385:       };
  386:       emit_state_change(&app, "error", Some(format!("spawn failed: {}", e)));
  387:       return Err(format!("spawn failed: {}", e));
  388:     }
  389:   };
  390: 
  391:   let stdin = child.stdin.take().ok_or("Failed to open stdin")?;
  392:   let stdout = child.stdout.take().ok_or("Failed to open stdout")?;
  393: 
  394:   let client = Arc::new(McpClient {
  395:     child: Mutex::new(child),
  396:     stdin: Mutex::new(stdin),
  397:     pending: Mutex::new(HashMap::new()),
  398:     next_id: AtomicU64::new(1),
  399:     command: cfg.command.clone(),
  400:     cwd: cfg.cwd.clone(),
  401:     _reader_abort: tokio::sync::Mutex::new(None),
  402:   });
  403: 
  404:   // Start reader loop
  405:   let c_out = client.clone();
  406:   let app_clone = app.clone();
  407:   let state_for_reader = Arc::new(Mutex::new(McpState::Disconnected));
  408:   {
  409:     let mut guard = state_for_reader.lock().await;
  410:     *guard = McpState::Connecting {
  411:       started_at: now_millis(),
  412:       cfg_summary: cfg_summary.clone(),
  413:     };
  414:   }
  415:   let reader_state = state_for_reader.clone();
  416:   
  417:   let reader_handle = tokio::spawn(async move {
  418:     reader_loop(stdout, c_out, app_clone, reader_state).await;
  419:   });
  420:   
  421:   // Store abort handle
  422:   {
  423:     let mut abort = client._reader_abort.lock().await;
  424:     *abort = Some(reader_handle.abort_handle());
  425:   }
  426: 
  427:   // Initialize with timeout
  428:   let init_result = timeout(
  429:     Duration::from_secs(INITIALIZATION_TIMEOUT_SECS),
  430:     request_with_client(
  431:       client.clone(),
  432:       "initialize",
  433:       json!({
  434:         "protocolVersion": "2024-11-05",
  435:         "clientInfo": { "name":"Nikolai Desktop", "version":"0.1.0" },
  436:         "capabilities": { "tools": {} }
  437:       }),
  438:     )
  439:   ).await;
  440: 
  441:   // Update the reader state reference
  442:   {
  443:     let mut rs = state_for_reader.lock().await;
  444:     *rs = McpState::Connected {
  445:       client: client.clone(),
  446:       connected_at: now_millis(),
  447:     };
  448:   }
  449: 
  450:   match init_result {
  451:     Ok(Ok(_)) => {
  452:       // Initialization successful
  453:       let _ = notify_with_client(client.clone(), "notifications/initialized", json!({})).await;
  454:       
  455:       // Transition to Connected
  456:       let mut guard = state.lock().await;
  457:       *guard = McpState::Connected {
  458:         client: client.clone(),
  459:         connected_at: now_millis(),
  460:       };
  461:       emit_state_change(&app, "connected", None);
  462:       
  463:       println!("[mcp] initialize success");
  464:       Ok(mcp_status().await)
  465:     }
  466:     Ok(Err(e)) => {
  467:       // Initialization failed with error
  468:       eprintln!("[mcp] initialization failed: {}", e);
  469:       
  470:       // Cleanup child
  471:       {
  472:         let mut child = client.child.lock().await;
  473:         let _ = child.start_kill();
  474:         let _ = timeout(Duration::from_secs(KILL_WAIT_TIMEOUT_SECS), child.wait()).await;
  475:       }
  476:       
  477:       // Abort reader
  478:       if let Some(abort) = client._reader_abort.lock().await.take() {
  479:         abort.abort();
  480:       }
  481:       
  482:       // Fail all pending
  483:       fail_all_pending(&client, "Initialization failed").await;
  484:       
  485:       // Transition to Error
  486:       let mut guard = state.lock().await;
  487:       *guard = McpState::Error {
  488:         message: format!("Initialization failed: {}", e),
  489:         at: now_millis(),
  490:       };
  491:       emit_state_change(&app, "error", Some(format!("Initialization failed: {}", e)));
  492:       
  493:       Err(format!("Initialization failed: {}", e))
  494:     }
  495:     Err(_elapsed) => {
  496:       // Initialization timed out
  497:       eprintln!("[mcp] initialization timed out");
  498:       
  499:       // Cleanup child
  500:       {
  501:         let mut child = client.child.lock().await;
  502:         let _ = child.start_kill();
  503:         let _ = timeout(Duration::from_secs(KILL_WAIT_TIMEOUT_SECS), child.wait()).await;
  504:       }
  505:       
  506:       // Abort reader
  507:       if let Some(abort) = client._reader_abort.lock().await.take() {
  508:         abort.abort();
  509:       }
  510:       
  511:       // Fail all pending
  512:       fail_all_pending(&client, "Initialization timed out").await;
  513:       
  514:       // Transition to Error
  515:       let mut guard = state.lock().await;
  516:       *guard = McpState::Error {
  517:         message: "Initialization timed out".to_string(),
  518:         at: now_millis(),
  519:       };
  520:       emit_state_change(&app, "error", Some("Initialization timed out".to_string()));
  521:       
  522:       Err("Initialization timed out".to_string())
  523:     }
  524:   }
  525: }

  527: #[tauri::command]
  528: pub async fn mcp_disconnect(app: AppHandle) -> Result<McpStatus, String> {
  529:   let state = ensure_state();
  530:   
  531:   // Transition state and extract client
  532:   let client_opt = {
  533:     let mut guard = state.lock().await;
  534:     let current = std::mem::replace(&mut *guard, McpState::Disconnected);
  535:     match current {
  536:       McpState::Connected { client, .. } => Some(client),
  537:       _ => None,
  538:     }
  539:   };
  540: 
  541:   emit_state_change(&app, "disconnected", None);
  542: 
  543:   if let Some(client) = client_opt {
  544:     // Abort reader task first
  545:     if let Some(abort) = client._reader_abort.lock().await.take() {
  546:       abort.abort();
  547:     }
  548:     
  549:     // Fail all pending requests
  550:     fail_all_pending(&client, "Disconnected").await;
  551:     
  552:     // Kill child and wait
  553:     let mut child = client.child.lock().await;
  554:     let _ = child.start_kill();
  555:     let _ = timeout(Duration::from_secs(KILL_WAIT_TIMEOUT_SECS), child.wait()).await;
  556:   }
  557: 
  558:   Ok(mcp_status().await)
  559: }
